import lowess from "@stdlib/stats-lowess";
import { Chart, registerables } from "chart.js";
import { sgg } from "ml-savitzky-golay-generalized";
import React, { useEffect, useState } from "react";
import { Line } from "react-chartjs-2";
import { getOrders } from "../../../../network/api";

Chart.register(...registerables);

/**
 * Represents a component for displaying a graph of total orders.
 * @module TotalOrdersGraph
 * @param {Object} props - The props object.
 * @param {Date} props.start_date - The start date for filtering orders.
 * @param {Date} props.end_date - The end date for filtering orders.
 * @param {string} props.smoothingOption - The option for applying a smoothing filter to the data.
 */
const TotalOrdersGraph = ({ start_date, end_date, smoothingOption }) => {
    /**
     * State to store the data for the order graph.
     * @type {[Object, function]}
     */
    const [orderData, setOrderData] = useState({ labels: [], datasets: [] });

    /**
     * Fetches data for the order graph based on the provided dates and smoothing option.
     * @function fetchData
     */
    useEffect(() => {
        const fetchData = async () => {
            try {
                const orders = await getOrders(start_date, end_date);
                const ordersPerDay = {};

                orders.forEach((order) => {
                    const orderDate = order[3].split(",")[1].trim();
                    if (!ordersPerDay[orderDate]) {
                        ordersPerDay[orderDate] = 0;
                    }
                    ordersPerDay[orderDate]++;
                });

                const orderDates = Object.keys(ordersPerDay);
                const orderCounts = Object.values(ordersPerDay);

                let yData;

                switch (smoothingOption) {
                    case "Savitzky-Golay Filter":
                        yData = sgg(orderCounts, 1, {
                            windowSize:
                                Math.floor(orderCounts.length / 20) % 2 === 0
                                    ? Math.floor(orderCounts.length / 20) + 1
                                    : Math.floor(orderCounts.length / 20),
                            derivative: 0,
                            polynomial: 3,
                        });
                        break;
                    case "LOWESS":
                        yData = lowess(
                            Array.from(
                                { length: orderCounts.length },
                                (_, i) => i + 1
                            ),
                            orderCounts,
                            { f: 0.1, nsteps: 10 }
                        ).y;
                        break;
                    default:
                        yData = orderCounts;
                        break;
                }

                const newOrderData = {
                    labels: orderDates,
                    datasets: [
                        {
                            label: "Number of Orders",
                            data: yData,
                            fill: false,
                            borderColor: "rgb(75, 192, 192)",
                        },
                    ],
                };

                setOrderData(newOrderData);
            } catch (error) {
                console.error("Error fetching order data:", error);
            }
        };

        // Fetch order data when component mounts or when start_date, end_date, or smoothingOption changes
        fetchData();
    }, [start_date, end_date, smoothingOption]);

    return (
        <div>
            <div className="chart-container">
                <h3>Orders Line Chart</h3>
                <Line
                    data={orderData}
                    options={{
                        scales: {
                            x: {
                                ticks: {
                                    autoSkip: true,
                                    maxRotation: 0,
                                    minRotation: 0,
                                    maxTicksLimit: 5,
                                },
                            },
                        },
                    }}
                />
            </div>
        </div>
    );
};

export default TotalOrdersGraph;
